				if (!uni.getStorageSync('token')) {

			},
			getindexList() {
				this.$request.http({
					url: "/api/index/index",
					method: 'post',
				}).then(res => {
					this.indexObj = res.data
				})
			},
			gethelpList() {
				this.$request.http({
					url: "/api/index/getPagesList",

			async grtuserinfo() {
				this.$request.http({
					url: '/api/member/userInfo',
					method: 'post',

				}).then(res => {
					console.log(res);
					if (res.code == 1) {
						this.userInfo = res.data
						uni.setStorageSync('userinfo', res.data)
						if (!this.token) {
							location.reload()

				})
			},
			async walletLogin(address, type = 1) {
				let res = await this.$request.http({
					url: '/api/login/login',
					method: 'post',
					data: {
						address: address,
						kind: type,
						invite_code: this.invite_code || '',
					}
				})
				uni.setStorageSync('token', res.data.token)
				uni.setStorageSync('userId', res.data.uid)
				this.grtuserinfo()
				// location.reload()
			},
			async walletLoginHandler() {
				await this.$nextTick();

				let provider = null;

				// ----- 钱包优先级顺序 -----
				if (window.bitkeep && window.bitkeep.ethereum) {
					provider = window.bitkeep.ethereum;

				} else if (window.okxwallet && window.okxwallet.ethereum) {
					provider = window.okxwallet.ethereum;

					// ====== SafePal 检测（重点）======
				} else if (window.safepal && window.safepal.ethereum) {
					provider = window.safepal.ethereum;

				} else if (window.ethereum && window.ethereum.isSafePal) {
					// SafePal 另一种注入方式
					provider = window.ethereum;

					// ====== TokenPocket（可选）=====
				} else if (window.ethereum && window.ethereum.isTokenPocket) {
					provider = window.ethereum;

					// ====== MetaMask 等通用钱包 ======
				} else if (window.ethereum) {
					provider = window.ethereum;
				}

				if (!provider) {
					console.log('未检测到钱包环境');
					return;
				}

				try {
					// 授权（必须在点击事件内调用）
					const accounts = await provider.request({
						method: 'eth_requestAccounts'
					});

					if (!accounts || !accounts.length) {
						console.log('用户未授权');
						return;
					}

					const address = accounts[0];
					this.address = address;

					// 你的登录逻辑
					this.walletLogin(address);
					this.showLogin = false;

				} catch (err) {
					console.error('钱包授权失败:', err);
				}
			},
			getAppWalletProvider() {
				// OKX Wallet (app)
				if (window.okxwallet?.ethereum) return window.okxwallet.ethereum;

				// Bitget / BitKeep
				if (window.bitkeep?.ethereum) return window.bitkeep.ethereum;

				// SafePal App
				if (window.safepal?.ethereum) return window.safepal.ethereum;
				if (window.ethereum?.isSafePal) return window.ethereum;

				// TokenPocket App
				if (window.ethereum?.isTokenPocket) return window.ethereum;

				// Trust Wallet (App Browser)
				if (window.trustwallet?.ethereum) return window.trustwallet.ethereum;

				// Coinbase App Browser
				if (window.ethereum?.isCoinbaseWallet) return window.ethereum;

				// imToken 内置浏览器
				if (/imtoken/i.test(navigator.userAgent)) {
					return window.ethereum;
				}

				// Binance Web3 Wallet App
				if (window.BinanceChain) return window.BinanceChain;

				// 通用：在钱包内置 DApp 浏览器中会注入 window.ethereum
				if (/wallet|token|okx|bitget|safepal|trust|coinbase/i.test(navigator.userAgent)) {
					if (window.ethereum) return window.ethereum;
				}

				return null;
			},
			// 登陆app钱包
			async loginByAppWallet_old() {
				const provider = this.getAppWalletProvider();

				if (!provider) {
					alert("Please open this page in the wallet's built-in browser.");
					return;
				}

				const accounts = await provider.request({
					method: "eth_requestAccounts"
				});
				
				this.walletLogin(accounts[0]);
			},
			async loginByAppWallet() {
				const injected = window.ethereum;

				// ========== 情况1：钱包内置浏览器 (Onchain、MetaMask、TP、OKX) ==========
				if (injected) {
					try {
						const accounts = await injected.request({
							method: "eth_requestAccounts"
						});
						return this.walletLogin(accounts[0]);
					} catch (e) {
						console.log("授权失败", e);
					}
				}

				// ========== 情况2：普通浏览器，自动拉起钱包 ==========
				this.openWalletApp();
			},
			async openWalletApp() {
			    try {
			        const result = await walletconnect.connect();
			        this.walletLogin(result.account);
			
			        uni.showToast({
			            title: "连接成功",
			            icon: "success"
			        });
			
			    } catch (error) {
			        console.error(error);
			        uni.showToast({
			            title: "连接失败",
			            icon: "none"
			        });
			    }
			},
			

			openWalletApp1() {
				const dappUrl = encodeURIComponent(window.location.href);

				const schemes = [
					// Onchain（伪示例，如需我可帮你确认官方 DeepLink）
					`onchainwallet://dapp?url=${dappUrl}`,

					// MetaMask
					`metamask://dapp/${window.location.host}`,

					// OKX Wallet
					`okx://wallet/dapp/url?url=${dappUrl}`,

					// TokenPocket
					`tpoutside://pull.activity?param={"url":"${dappUrl}"}`,

					// Bitget Wallet
					`bitkeep://dapp?url=${dappUrl}`,

					// OneKey Wallet
					`onekeywallet://dapp?url=${dappUrl}`

					// TokenPocket（TP）
					`tpoutside://pull.activity?param={"url":"${dappUrl}"}`,
				];

				// 多方案容错，只要一个能打开就行
				for (let link of schemes) {
					let iframe = document.createElement("iframe");
					iframe.style.display = "none";
					iframe.src = link;
					document.body.appendChild(iframe);

					setTimeout(() => {
						document.body.removeChild(iframe);
					}, 2000);
				}

				// 兜底提示
				setTimeout(() => {
					alert("请安装支持以太坊的钱包 App 并重试");
				}, 2500);
			},

			getBrowserWalletProvider() {
				if (window.ethereum) return window.ethereum;
				return null;
			},
			// 登陆浏览器钱包
			async loginByBrowserWallet() {

					return;

				const accounts = await provider.request({
					method: "eth_requestAccounts"
				});

				this.walletLogin(accounts[0]);
			},

			async connectSafePalWallet() {
				let provider = null;

				// SafePal 注入方式 1
				if (window.safepal && window.safepal.ethereum) {
					provider = window.safepal.ethereum;

					provider = window.ethereum;
				}

				if (!provider) {
					console.log("未检测到 SafePal 钱包环境");
					return;
				}

				try {
					const accounts = await provider.request({
						method: "eth_requestAccounts"

					if (!accounts.length) {
						console.log("SafePal 钱包未授权");
						return;

					const address = accounts[0];

				} catch (err) {

export default {

    async connect() {
        return new Promise(async (resolve, reject) => {

            try {
                if (!window.WalletConnectProvider) {
                    return reject("未加载 WalletConnectProvider 资源");
                }

                wcProvider = await window.WalletConnectProvider.init({
                    projectId,
                    metadata,
                    showQrModal: true,        // 支持扫码
                    optionalChains: [1, 56],  // ETH, BSC
                    optionalMethods: ["eth_sendTransaction", "personal_sign"]
                });

                await wcProvider.connect();

                const accounts = wcProvider.accounts;

                resolve({
                    account: accounts[0],
                    provider: wcProvider
                });

            } catch (e) {
                reject(e);
            }
        });
    },

    disconnect() {
        if (wcProvider) {
            wcProvider.disconnect();
            wcProvider = null;
        }
    }
